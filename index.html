// 감독교사용 QR코드 발급 function downloadTeacherQR() { let baseUrl = window.location.href.split('index.html')[0]; if (!baseUrl.endsWith('/')) baseUrl += '/'; let loginUrl = `${baseUrl}scan.html`; // 파라미터가 없으면 로그인 화면 오픈 const { jsPDF } = window.jspdf; const pdf = new jsPDF('p', 'mm', 'a4'); const qrDiv = document.createElement('div'); new QRCode(qrDiv, { text: loginUrl, width: 256, height: 256, colorDark: "#1976d2", correctLevel: QRCode.CorrectLevel.H }); setTimeout(() => { const canvas = qrDiv.querySelector('canvas'); pdf.setFont("Noto Sans KR", "bold"); pdf.setFontSize(20); pdf.text("👨‍🏫 감독교사 스캐너 로그인용 QR코드", 105, 40, {align: 'center'}); pdf.addImage(canvas.toDataURL('image/png'), 'PNG', 55, 60, 100, 100); pdf.save('감독교사_전용QR.pdf'); }, 100); }
